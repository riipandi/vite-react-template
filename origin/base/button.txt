export type ButtonStyles = VariantProps<typeof buttonStyles>

export interface ButtonProps extends BaseButtonProps, ButtonStyles {
  className?: string | ((state: ButtonState) => string | undefined)
}

export function Button({
  color,
  variant,
  size,
  mode,
  pill,
  progress,
  block,
  className,
  ...props
}: ButtonProps) {
  const styles = buttonStyles({ color, variant, size, mode, pill, progress, block })
  return (
    <BaseButton
      data-slot='button'
      data-size={size}
      data-mode={mode}
      className={(state) =>
        clx(styles, typeof className === 'function' ? className(state) : className)
      }
      focusableWhenDisabled
      {...props}
    />
  )
}
